    <script>

        function closeDeleteBox() {
            //cập nhật cái deleteBox: ẩn nó
            document.getElementById("deleteBox").style.display="none";
        }

        function openDeleteBox(link) {
            const id = link.getAttribute("data-id");
            const name = link.getAttribute("data-name");

            //cập nhật cái deleteBox: show nó
            document.getElementById("deleteBox").style.display="block";

            //cập nhật cái deleteBox: show message động...
            document.getElementById("deleteMsg").innerText =
                     "Do you really want to delete product " + id + " - " + name + "?";

            //cập nhật cái hành động trong form
            document.getElementById("deleteForm").action = "/dieudao/fruits/delete/" + id;
        }
    </script>
